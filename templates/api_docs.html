{{ _('Calling methods') }}

{# ---------------- Normal (synchronous) ---------------- #}

{{ _('Synchronous request that waits for the result. Simplest to use.') }}

{{ _('Text detection') }}

curl -X POST "{{ base }}/api/v1/detect" \
  -H "X-API-Key: YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"text":"some text","scene":"message"}'

{{ _('Optional') }}: scene (message/nickname), categories, prompt_id. {{ _('Returns') }} {"result": bool} ({{ _('true = violation') }}).

{{ _('Image detection') }}

curl -X POST "{{ base }}/api/v1/detect" \
  -H "X-API-Key: YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"media_type":"image","image_url":"https://example.com/a.jpg"}'

{{ _('Only public image URLs are accepted.') }} {{ _('Returns') }} {"result": bool, "label": "..."}

{{ _('Video detection') }}

curl -X POST "{{ base }}/api/v1/detect" \
  -H "X-API-Key: YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"media_type":"video","video_url":"https://example.com/a.mp4"}'

{{ _('Video requires a channel/model with native video support (e.g. Gemini).') }}

{# ---------------- Polling ---------------- #}

{{ _('Submit a task then poll for the result. Avoids timeouts on long checks.') }}

1. {{ _('Submit a detection task') }} — {{ _('Text detection') }}

curl -X POST "{{ base }}/api/v1/detect/submit" \
  -H "X-API-Key: YOUR_KEY" -H "Content-Type: application/json" \
  -d '{"text":"some text"}'

1. {{ _('Submit a detection task') }} — {{ _('Image detection') }}

curl -X POST "{{ base }}/api/v1/detect/submit" \
  -H "X-API-Key: YOUR_KEY" -H "Content-Type: application/json" \
  -d '{"media_type":"image","image_url":"https://example.com/a.jpg"}'

1. {{ _('Submit a detection task') }} — {{ _('Video detection') }}

curl -X POST "{{ base }}/api/v1/detect/submit" \
  -H "X-API-Key: YOUR_KEY" -H "Content-Type: application/json" \
  -d '{"media_type":"video","video_url":"https://example.com/a.mp4"}'

2. {{ _('Poll for the result') }}

curl "{{ base }}/api/v1/detect/result?task_id=TASK_ID" \
  -H "X-API-Key: YOUR_KEY"

{{ _('While processing:') }} {"status":"pending"}; {{ _('when done:') }} {"status":"done","result":{...}}
